        function typeMessage(elementId, messages, delay = 1000) {
            let elem = document.getElementById(elementId);
            let index = 0;
            let charIndex = 0;
            function type() {
                if (charIndex < messages[index].length) {
                    elem.textContent += messages[index][charIndex];
                    charIndex++;
                    setTimeout(type, 100);
                } else {
                    setTimeout(() => {
                        elem.textContent = "";
                        index = (index + 1) % messages.length;
                        charIndex = 0;
                        type();
                    }, delay);
                }
            }
            type();
        }
        typeMessage("animatedText", ["Hi!", "I'm Abhishek."], 1000);
    </script>
